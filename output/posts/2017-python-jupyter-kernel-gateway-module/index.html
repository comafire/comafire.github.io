class Store:
    store = None
    
    def __init__(self):
        self.store = {}
    
    def post(self, req):
        user_id = req["path"]["user_id"]
        name = req["body"]["name"]
        res = {}
  
        self.store[user_id] = name
        res["status"] = "SUCCESS"
        res["name"] = name
        return json.dumps(res)

    def get(self, req):
        user_id = req["path"]["user_id"]
        res = {}
        
        name = self.store[user_id]
        res["status"] = "SUCCESS"
        res["name"] = name
        return json.dumps(res)
    
    def delete(self, req):
        user_id = req["path"]["user_id"]
        res = {}
        
        del self.store[user_id]
        return json.dumps(res)
